#pragma warning (disable:4996)
#include <iostream>
#include <vector>
#include <string.h>
#define MAX 510

using namespace std;

int v, e;
int Degrees[MAX];

bool Judge(int num){
	if (num % 2 == 0)
		return true;
	else
		return false;
}

int main(){
	memset(Degrees, 0, sizeof(Degrees));
	scanf("%d%d", &v, &e);

	for (int i = 0; i < e; i++){
		int e1, e2;
		scanf("%d%d", &e1, &e2);
		Degrees[e1]++;
		Degrees[e2]++;
	}
	
	int num_odd = 0;
	for (int i = 1; i <= v; i++){
		if (i == v)
			printf("%d\n", Degrees[i]);
		else 
			printf("%d ", Degrees[i]);
		if (!Judge(Degrees[i]))
			//ÆæÊý
			num_odd++;
	}

	if (num_odd == 0)
		printf("Eulerian\n");

	else if (num_odd == 2)
		printf("Semi-Eulerian\n");
	else
		printf("Non-Eulerian\n");

	system("pause");
	return 0;
}
